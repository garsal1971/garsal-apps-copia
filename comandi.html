#001
Crea tabella cm_rewards_log su Supabase
Necessaria per il modulo Premi: registra ogni riscatto (premio ritirato, punti spesi, data). Esegui la query nell'SQL Editor di Supabase → dashboard.supabase.com → progetto garsal-apps → SQL Editor.
⏳ da fare SQL Supabase 
-- Tabella log riscatti premi
create table if not exists cm_rewards_log (
  id           uuid        default gen_random_uuid() primary key,
  reward_id    uuid        references cm_rewards(id) on delete set null,
  reward_name  text        not null,
  points_spent integer     not null check (points_spent > 0),
  redeemed_at  timestamptz default now() not null
);

-- Indice per query cronologia (ordinate per data desc)
create index if not exists cm_rewards_log_redeemed_at_idx
  on cm_rewards_log (redeemed_at desc);

-- RLS: abilita e consenti solo utenti autenticati
alter table cm_rewards_log enable row level security;

create policy "Auth users can read their log"
  on cm_rewards_log for select
  using (auth.role() = 'authenticated');

create policy "Auth users can insert"
  on cm_rewards_log for insert
  with check (auth.role() = 'authenticated');
📍 Dove: Supabase Dashboard → SQL Editor → New Query → incolla → Run
#002
Aggiungi colonne is_redeemed e use_count a cm_rewards
Necessarie per la logica post-riscatto: i premi una tantum vengono nascosti dalla lista dopo il primo riscatto; i premi ripetibili tracciano il numero di utilizzi e il costo si aggiorna ad ogni riscatto (points_cost += points_per_use). Esegui nell'SQL Editor di Supabase.
⏳ da fare SQL Supabase 
-- Colonna per marcare premi "una tantum" già riscattati
alter table cm_rewards
  add column if not exists is_redeemed boolean not null default false;

-- Colonna contatore utilizzi per premi "ripetibili"
alter table cm_rewards
  add column if not exists use_count integer not null default 0;

-- Verifica
select id, name, type, points_cost, points_per_use, is_redeemed, use_count
from cm_rewards
order by created_at;
📍 Dove: Supabase Dashboard → SQL Editor → New Query → incolla → Run
