-- ── SHARED ──────────────────────────────────────────────────────
CREATE TABLE IF NOT EXISTS cm_apps (
  id            SERIAL PRIMARY KEY,
  title         TEXT NOT NULL,
  description   TEXT,
  html_file     TEXT NOT NULL,
  score_query   TEXT,
  color         TEXT,
  active        BOOLEAN DEFAULT true,
  riservato     BOOLEAN DEFAULT false
);

CREATE TABLE IF NOT EXISTS cm_categories (
  id    SERIAL PRIMARY KEY,
  name  TEXT NOT NULL,
  color TEXT,
  icon  TEXT
);

CREATE TABLE IF NOT EXISTS cm_settings (
  key   TEXT PRIMARY KEY,
  value TEXT
);

CREATE TABLE IF NOT EXISTS cm_notification_rules (
  id          SERIAL PRIMARY KEY,
  app         TEXT,
  entity_id   TEXT,
  rule_type   TEXT,
  fire_at     TIMESTAMPTZ,
  payload     JSONB,
  active      BOOLEAN DEFAULT true,
  created_at  TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE IF NOT EXISTS cm_notification_queue (
  id          SERIAL PRIMARY KEY,
  app         TEXT,
  entity_id   TEXT,
  fire_at     TIMESTAMPTZ,
  status      TEXT DEFAULT 'pending',
  payload     JSONB,
  metadata    JSONB,
  created_at  TIMESTAMPTZ DEFAULT now()
);

-- ── TASKS ───────────────────────────────────────────────────────
CREATE TABLE IF NOT EXISTS ts_tasks (
  id                    SERIAL PRIMARY KEY,
  user_id               UUID,
  title                 TEXT NOT NULL,
  type                  TEXT DEFAULT 'single',
  status                TEXT DEFAULT 'active',
  priority_id           INT,
  category_id           INT,
  start_date            TIMESTAMPTZ,
  next_occurrence_date  TIMESTAMPTZ,
  recurrence_rule       JSONB,
  notes                 TEXT,
  created_at            TIMESTAMPTZ DEFAULT now(),
  updated_at            TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE IF NOT EXISTS ts_history (
  id          SERIAL PRIMARY KEY,
  task_id     INT REFERENCES ts_tasks(id) ON DELETE CASCADE,
  action      TEXT CHECK (action IN (
                'completed','completed_late','failed','terminated',
                'archived','skipped','reactivated','rollback'
              )),
  completed_at TIMESTAMPTZ DEFAULT now(),
  notes       TEXT
);

CREATE TABLE IF NOT EXISTS ts_priorities (
  id     SERIAL PRIMARY KEY,
  name   TEXT NOT NULL,
  color  TEXT,
  weight INT DEFAULT 0
);

CREATE TABLE IF NOT EXISTS ts_settings (
  key   TEXT PRIMARY KEY,
  value TEXT
);

CREATE TABLE IF NOT EXISTS ts_notes (
  id         SERIAL PRIMARY KEY,
  title      TEXT,
  content    TEXT,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- ── HABIT TRACKER ────────────────────────────────────────────────
CREATE TABLE IF NOT EXISTS hb_habits (
  id          SERIAL PRIMARY KEY,
  user_id     UUID,
  name        TEXT NOT NULL,
  stack_order INT DEFAULT 0,
  points      INT DEFAULT 10,
  active      BOOLEAN DEFAULT true,
  created_at  TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE IF NOT EXISTS hb_completions (
  id           SERIAL PRIMARY KEY,
  habit_id     INT REFERENCES hb_habits(id) ON DELETE CASCADE,
  completed_on DATE NOT NULL,
  created_at   TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE IF NOT EXISTS hb_user_points (
  user_id    UUID PRIMARY KEY,
  balance    INT DEFAULT 0,
  updated_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE IF NOT EXISTS hb_points_transactions (
  id         SERIAL PRIMARY KEY,
  user_id    UUID,
  delta      INT NOT NULL,
  reason     TEXT,
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE IF NOT EXISTS hb_archived_stacks (
  id          SERIAL PRIMARY KEY,
  user_id     UUID,
  stack_data  JSONB,
  archived_at TIMESTAMPTZ DEFAULT now()
);

-- ── EVENTS LOG ──────────────────────────────────────────────────
CREATE TABLE IF NOT EXISTS el_groups (
  id    SERIAL PRIMARY KEY,
  name  TEXT NOT NULL,
  icon  TEXT,
  color TEXT
);

CREATE TABLE IF NOT EXISTS el_events (
  id       SERIAL PRIMARY KEY,
  group_id INT REFERENCES el_groups(id) ON DELETE CASCADE,
  name     TEXT NOT NULL,
  icon     TEXT,
  color    TEXT
);

CREATE TABLE IF NOT EXISTS el_logs (
  id         SERIAL PRIMARY KEY,
  event_id   INT REFERENCES el_events(id) ON DELETE CASCADE,
  logged_at  TIMESTAMPTZ DEFAULT now(),
  notes      TEXT
);

-- ── WEIGHT QUEST ────────────────────────────────────────────────
CREATE TABLE IF NOT EXISTS ps_weight_tracking (
  id          SERIAL PRIMARY KEY,
  user_id     UUID,
  weight_kg   NUMERIC(5,2) NOT NULL,
  measured_at DATE NOT NULL DEFAULT CURRENT_DATE,
  notes       TEXT,
  created_at  TIMESTAMPTZ DEFAULT now()
);

-- ── RPC run_score_query (usata dal launcher) ────────────────────
CREATE OR REPLACE FUNCTION run_score_query(query_sql TEXT)
RETURNS INT LANGUAGE plpgsql SECURITY DEFINER AS $$
DECLARE result INT;
BEGIN
  EXECUTE query_sql INTO result;
  RETURN COALESCE(result, 0);
END;
$$;

▶ 3 · netlify — abilita branch deploys

Abilita branch deploys per dev/* su Netlify
Ogni push su un branch dev/ deve generare un URL preview Netlify.
L'URL avrà il formato: dev--nome-branch--<sitename>.netlify.app

Come usare l'ambiente dev

riferimento 
## SVILUPPO (branch dev)
git checkout -b dev/mia-feature
# modifica i file HTML
git add .
git commit -m "feat: descrizione modifica"
git push origin dev/mia-feature
# → Netlify crea preview: dev--mia-feature--.netlify.app
# → Le app usano automaticamente il DB Supabase DEV

## PRODUZIONE (branch claude/**)
git checkout -b claude/mia-feature-XXXXX
# modifica i file HTML (incrementa APP_VERSION e BUILD_TIME)
git add .
git commit -m "feat: descrizione modifica"
git push origin claude/mia-feature-XXXXX
# → GitHub Actions merge su master → Netlify deploya in produzione
# → Le app usano il DB Supabase PROD
📍 Le credenziali Supabase vengono selezionate automaticamente in base all'URL del browser
