CREATE TABLE cm_profile — profilo personale utente
Crea la tabella cm_profile con dati anagrafici e clinici (patologie, farmaci). Un solo record per utente (UNIQUE su user_id). Include RLS e trigger updated_at.
SQL PROD cm_profile 
CREATE TABLE IF NOT EXISTS cm_profile (
  id            uuid        PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id       uuid        NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,

  -- Anagrafica
  nome          text,
  cognome       text,
  email         text,
  avatar_url    text,
  data_nascita  date,
  sesso         text        CHECK (sesso IN ('M', 'F', 'Altro')),
  altezza_cm    smallint    CHECK (altezza_cm > 50 AND altezza_cm < 300),

  -- Clinica
  patologie     text,       -- testo libero, una patologia per riga
  farmaci       text,       -- testo libero, un farmaco per riga
  note_mediche  text,

  -- Meta
  created_at    timestamptz NOT NULL DEFAULT now(),
  updated_at    timestamptz NOT NULL DEFAULT now(),

  UNIQUE (user_id)
);

-- Trigger updated_at
CREATE OR REPLACE FUNCTION cm_profile_set_updated_at()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$;

CREATE TRIGGER cm_profile_updated_at
  BEFORE UPDATE ON cm_profile
  FOR EACH ROW EXECUTE FUNCTION cm_profile_set_updated_at();

-- RLS
ALTER TABLE cm_profile ENABLE ROW LEVEL SECURITY;

CREATE POLICY "profilo: lettura proprio record"
  ON cm_profile FOR SELECT
  USING (auth.uid() = user_id);

CREATE POLICY "profilo: inserimento proprio record"
  ON cm_profile FOR INSERT
  WITH CHECK (auth.uid() = user_id);

CREATE POLICY "profilo: aggiornamento proprio record"
  ON cm_profile FOR UPDATE
  USING (auth.uid() = user_id);

CREATE POLICY "profilo: eliminazione proprio record"
  ON cm_profile FOR DELETE
  USING (auth.uid() = user_id);

-- GRANT (obbligatorio: senza questi l'RLS blocca tutto)
GRANT ALL ON TABLE cm_profile TO authenticated;
GRANT ALL ON TABLE cm_profile TO service_role;

FIX — RLS cm_profile (esegui questo se hai già la tabella)
Risolve "new row violates row-level security policy".
Aggiunge GRANT, imposta DEFAULT auth.uid() su user_id e ricrea le policy RLS in modo sicuro.
SQL FIX cm_profile 
-- 1. GRANT ruolo authenticated
GRANT ALL ON TABLE cm_profile TO authenticated;
GRANT ALL ON TABLE cm_profile TO service_role;

-- 2. user_id si imposta automaticamente dal JWT (non serve mandarlo dal frontend)
ALTER TABLE cm_profile ALTER COLUMN user_id SET DEFAULT auth.uid();

-- 3. Ricrea policy RLS (DROP + CREATE per sicurezza)
DROP POLICY IF EXISTS "profilo: lettura proprio record"    ON cm_profile;
DROP POLICY IF EXISTS "profilo: inserimento proprio record" ON cm_profile;
DROP POLICY IF EXISTS "profilo: aggiornamento proprio record" ON cm_profile;
DROP POLICY IF EXISTS "profilo: eliminazione proprio record" ON cm_profile;

CREATE POLICY "profilo: lettura proprio record"
  ON cm_profile FOR SELECT TO authenticated
  USING (auth.uid() = user_id);

CREATE POLICY "profilo: inserimento proprio record"
  ON cm_profile FOR INSERT TO authenticated
  WITH CHECK (auth.uid() = user_id);

CREATE POLICY "profilo: aggiornamento proprio record"
  ON cm_profile FOR UPDATE TO authenticated
  USING (auth.uid() = user_id);

CREATE POLICY "profilo: eliminazione proprio record"
  ON cm_profile FOR DELETE TO authenticated
  USING (auth.uid() = user_id);

Questo script assegna il tuo user_id a tutti i record orfani.
Sostituisci TUO-USER-UUID con il tuo UUID da Authentication → Users → copia UUID.

-- Sostituisci TUO-USER-UUID con il tuo UUID reale
-- (Authentication → Users → clic sull'utente → copia UUID)

UPDATE ps_objectives
  SET user_id = 'TUO-USER-UUID'
  WHERE user_id IS NULL;

UPDATE ps_weight_tracking
  SET user_id = 'TUO-USER-UUID'
  WHERE user_id IS NULL;
