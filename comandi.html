#003
Snippet Java/JDBC per chiamare KCBPANAG
Codice Java pronto all'uso con CallableStatement. Imposta i 4 parametri IN, registra i 25 OUT, esegue la CALL e legge i risultati. Adattare i valori di input e il DataSource al proprio contesto.
▶ snippet Java JDBC 
String sql = "{CALL DB2C.KCBPANAG(?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?)}";

try (CallableStatement cs = conn.prepareCall(sql)) {

    // --- parametri IN (1-4) ---
    cs.setString(1, cdInterno);   // CD_INTERNO  CHAR(5)
    cs.setString(2, ndg);         // NDG         CHAR(16)
    cs.setString(3, lingua);      // LINGUA      CHAR(2)
    cs.setString(4, country);     // COUNTRY     CHAR(2)

    // --- parametri OUT (5-29) ---
    cs.registerOutParameter( 5, Types.CHAR);  // COMPANY_NAME
    cs.registerOutParameter( 6, Types.CHAR);  // REG_ADDRESS
    cs.registerOutParameter( 7, Types.CHAR);  // TOWN
    cs.registerOutParameter( 8, Types.CHAR);  // POSTAL_CODE
    cs.registerOutParameter( 9, Types.CHAR);  // COUNTRY_HO
    cs.registerOutParameter(10, Types.CHAR);  // COUNTRY_BRANCH
    cs.registerOutParameter(11, Types.CHAR);  // CUSTOMER_COUNT_1
    cs.registerOutParameter(12, Types.CHAR);  // ROLE_1
    cs.registerOutParameter(13, Types.CHAR);  // E_MAIL_1
    cs.registerOutParameter(14, Types.CHAR);  // FAX_NUMB_1
    cs.registerOutParameter(15, Types.CHAR);  // CUSTOMER_COUNT_2
    cs.registerOutParameter(16, Types.CHAR);  // ROLE_2
    cs.registerOutParameter(17, Types.CHAR);  // E_MAIL_2
    cs.registerOutParameter(18, Types.CHAR);  // FAX_NUMB_2
    cs.registerOutParameter(19, Types.CHAR);  // SWIFT_BIC
    cs.registerOutParameter(20, Types.CHAR);  // UIC
    cs.registerOutParameter(21, Types.CHAR);  // TYPE_HBS
    cs.registerOutParameter(22, Types.CHAR);  // PURPOSE_BUSN
    cs.registerOutParameter(23, Types.CHAR);  // EB_VOLUME_MONTH
    cs.registerOutParameter(24, Types.CHAR);  // EB_REVENUES_YEAR
    cs.registerOutParameter(25, Types.CHAR);  // ADDITIONAL_INFO
    cs.registerOutParameter(26, Types.CHAR);  // TP_REL
    cs.registerOutParameter(27, Types.CHAR);  // FLAG_NDG_POTENZ
    cs.registerOutParameter(28, Types.CHAR);  // RET_CODE
    cs.registerOutParameter(29, Types.CHAR);  // MSG

    cs.execute();

    // --- leggi esito ---
    String retCode = cs.getString(28); // es. "00" = OK
    String msg     = cs.getString(29);

    if (!"00".equals(retCode != null ? retCode.trim() : "")) {
        throw new RuntimeException("KCBPANAG errore: " + retCode + " — " + msg);
    }

    // --- leggi dati output ---
    String companyName = cs.getString(5);
    String address     = cs.getString(6);
    String town        = cs.getString(7);
    // ... ecc.
}

#006
Diagnosi -440: CALL da connessione corretta + CURRENT PATH
Cause probabili ora che i permessi sono OK:
1. La CALL va sulla connessione emergenza loc0d0 ma la procedura risiede nel subsystem/location a cui punta loujd0 testkcb (tab 2)
2. CURRENT PATH non include DB2C (raro con schema esplicito ma verificare)
3. I ? nella CALL non sono gestiti come OUT dal tool — usare Ctrl+Enter solo sulla CALL (non F5 sull'intero script)

KcbpanagCaller.java — classe completa pronta all'uso
import java.sql.*;

public class KcbpanagCaller {

    // ── CONFIGURAZIONE ─────────────────────────────────────────
    // Formato z/OS:  jdbc:db2://HOST:PORT/LOCATION
    // Formato LUW:   jdbc:db2://HOST:PORT/DATABASE
    static final String URL  = "jdbc:db2://HOSTNAME:50000/TESTKCB";
    static final String USER = "loujd0";
    static final String PASS = "password";
    // ───────────────────────────────────────────────────────────

    public static void main(String[] args) throws Exception {

        // Parametri IN — adattare ai valori reali
        String cdInterno = "02008";
        String ndg       = "0000000024159636";
        String lingua    = "IT";
        String country   = "IT";

        String sql = "{CALL DB2C.KCBPANAG(?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?)}";

        try (Connection conn = DriverManager.getConnection(URL, USER, PASS);
             CallableStatement cs = conn.prepareCall(sql)) {

            // IN
            cs.setString(1, cdInterno);
            cs.setString(2, ndg);
            cs.setString(3, lingua);
            cs.setString(4, country);

            // OUT
            for (int i = 5; i <= 29; i++) {
                cs.registerOutParameter(i, Types.CHAR);
            }

            cs.execute();

            // Esito
            String retCode = cs.getString(28);
            String msg     = cs.getString(29);
            System.out.println("RET_CODE : [" + retCode + "]");
            System.out.println("MSG      : [" + msg     + "]");

            // Dati
            System.out.println("COMPANY_NAME : " + cs.getString(5));
            System.out.println("REG_ADDRESS  : " + cs.getString(6));
            System.out.println("TOWN         : " + cs.getString(7));
            System.out.println("POSTAL_CODE  : " + cs.getString(8));
            System.out.println("COUNTRY_HO   : " + cs.getString(9));
            System.out.println("SWIFT_BIC    : " + cs.getString(19));
            System.out.println("TP_REL       : " + cs.getString(26));
            System.out.println("FLAG_NDG_POT : " + cs.getString(27));
        }
    }
}

Errori comuni e soluzioni
// Errore: "No suitable driver found"
//   → db2jcc4.jar non è nel classpath, oppure URL sbagliato

// Errore: "Communication link failure" / ERRORCODE=-4499
//   → HOST o PORT errati; DB2 non raggiungibile dalla macchina
//   → verificare: telnet HOSTNAME 50000

// Errore: SQLCODE -440 anche da Java
//   → problema di CURRENT PATH / location
//   → aggiungere dopo la connessione:
conn.setSchema("DB2C");  // oppure:
try (Statement st = conn.createStatement()) {
    st.execute("SET CURRENT PATH = SYSIBM, SYSFUN, SYSPROC, DB2C");
}

// Errore: SQLCODE -30082 (autenticazione)
//   → credenziali errate o utente non autorizzato alla connessione JDBC
📍 URL di connessione: chiedere al DBA l'hostname, la porta (default 50000) e il nome del subsystem/location DB2 z/OS · Il JAR db2jcc4.jar è la versione universale (supporta sia z/OS che LUW) — preferirla a db2jcc.jar
